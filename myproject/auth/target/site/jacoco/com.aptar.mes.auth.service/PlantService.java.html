package com.aptar.mes.auth.service;

import java.util.HashMap;
import java.util.List;
import java.util.Optional;

import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.stereotype.Service;

import com.aptar.mes.auth.repository.ConfigurationRepository;
import com.aptar.mes.auth.repository.OAuthUserRepository;
import com.aptar.mes.auth.repository.PlantRepository;
import com.aptar.mes.auth.service.dto.PlantDTO;
import com.aptar.mes.auth.service.dto.UserPlantsDTO;
import com.aptar.mes.core.general.db.entity.ConfigurationEntity;
import com.aptar.mes.core.general.db.entity.OAuthUserEntity;
import com.aptar.mes.core.general.db.entity.PlantEntity;

import lombok.extern.java.Log;

@Service
@Log
/**
 * This class implements the logic for Plant
 *
 */

public class PlantService {

	@Autowired
	private OAuthUserRepository userRepository;

	@Autowired
	private PlantRepository plantRepository;
	
	@Autowired
	private ConfigurationRepository configurationRepository;

	/**
	 * Return plant list associated with the user
	 * 
	 * @return list associated with the user
	 */
	public UserPlantsDTO getPlantList(String username) {

		log.fine("Finding entities on DB");
		OAuthUserEntity user = userRepository.findByUsername(username);
		UserPlantsDTO userPlantsDTO = new UserPlantsDTO();

		if(user != null) {

			for (PlantEntity plant: user.getPlants()) {
				userPlantsDTO.getPlants().add(new PlantDTO(plant.getPlantId(), plant.getDescription()));
			}

			//set default plant if it's present and if it's included in plants
			if(user.getPlantId() != null && !userPlantsDTO.getPlants().isEmpty()) {

				Optional<PlantEntity> plantEntity = plantRepository.findById(user.getPlantId());
				if(plantEntity.isPresent()) {
					PlantDTO plantDTO = new PlantDTO(plantEntity.get().getPlantId(), plantEntity.get().getDescription());
					if(userPlantsDTO.getPlants().contains(plantDTO)) {
						userPlantsDTO.setDefaultPlant(plantDTO);
					}
				}
			}
		}

		return userPlantsDTO;

	}
	
	public HashMap<String, String> getPlantConfigurations(String plantId) {
		
		HashMap<String, String> configurationsMap = new HashMap<>();
		List<ConfigurationEntity> configurations = configurationRepository.findByPlantId(plantId);
		
		for(ConfigurationEntity config : configurations) {
			configurationsMap.put(config.getConfigurationName(), config.getCfgValue());
		}
		
		return configurationsMap;
	}
}
