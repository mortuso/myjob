package com.aptar.mes.auth.service;

import java.util.List;

import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.stereotype.Service;

import com.aptar.mes.auth.repository.ConfigurationRepository;
import com.aptar.mes.auth.service.dto.SeparatorsDTO;
import com.aptar.mes.core.general.db.entity.ConfigurationEntity;

import lombok.extern.java.Log;

@Service
@Log
/**
 * This class implements the logic for configuration
 *
 */
public class SeparatorService {

	@Autowired
	private ConfigurationRepository repository;

	public static final String SEPARATOR_CONFIG_KEY = "CFG_SEPARATOR"; 
	
	/**
	 * Return all the separator values on configuration
	 * @return list of all separators
	 */
	public SeparatorsDTO getSeparators() {

		log.fine("Finding entities on DB");
		List<ConfigurationEntity> configurations = repository.findByConfigurationName(SEPARATOR_CONFIG_KEY);
		SeparatorsDTO separatorDTO = new SeparatorsDTO();
		
		for (ConfigurationEntity configuration:configurations){
			separatorDTO.getValues().add(configuration.getCfgValue());
		}
		
		return separatorDTO;
	}
}
