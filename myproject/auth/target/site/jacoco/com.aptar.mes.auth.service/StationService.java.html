package com.aptar.mes.auth.service;

import java.util.List;

import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.stereotype.Service;

import com.aptar.mes.auth.repository.OAuthUserRepository;
import com.aptar.mes.auth.repository.StationRepository;
import com.aptar.mes.auth.service.dto.StationDTO;
import com.aptar.mes.auth.service.dto.UserStationsDTO;
import com.aptar.mes.common.exception.ExceptionInfo;
import com.aptar.mes.common.exception.MesApplicationException;
import com.aptar.mes.core.general.db.entity.OAuthUserEntity;
import com.aptar.mes.core.general.db.entity.StationEntity;

import lombok.Getter;
import lombok.extern.java.Log;

@Service
@Log
/**
 * This class implements the logic for Station
 *
 */
public class StationService {

	@Autowired
	StationRepository stationRepository;

	@Autowired
	OAuthUserRepository userRepository;


	/**
	 * Return station list associated with the user and plant
	 * 
	 * @return list associated with the user and plant
	 */
	public UserStationsDTO getStationList(String plantId, String username, boolean isEnabledChangeStation) {

		log.fine("Finding entities on DB");

		UserStationsDTO userStationsDTO = new UserStationsDTO();

		OAuthUserEntity user = userRepository.findByUsername(username);

		if(user != null) {

			List<StationEntity> stations = stationRepository.findByPlantId(plantId);

			// set stations
			for(StationEntity station: stations) {
				StationDTO stationDTO = new StationDTO();
				stationDTO.setId(station.getStationPk());
				stationDTO.setDescription(station.getName());
				userStationsDTO.getStations().add(stationDTO);
			}

			//set default station if it's present and if it's included in stations
			if(user.getStation() != null) {
				StationDTO defaultStation = new StationDTO();
				defaultStation.setId(user.getStation().getStationPk());
				defaultStation.setDescription(user.getStation().getName());
				if(userStationsDTO.getStations().contains(defaultStation)) {
					userStationsDTO.setDefaultStation(defaultStation);
				}
			}

			// view stations only if user is enabled to change station (show default station if it's present)
			if(!isEnabledChangeStation && !userStationsDTO.getStations().isEmpty()) {
				// show default station in avalaible station
				if(userStationsDTO.getDefaultStation() != null) {
					userStationsDTO.getStations().removeIf(e -> e.getId() != userStationsDTO.getDefaultStation().getId());
				} else {
					userStationsDTO.getStations().clear();
				}
			}

			// User doesn't have a default station and is not enabled to change stations
			checkStation(isEnabledChangeStation, userStationsDTO);
		}
		return userStationsDTO;
	}

	private void checkStation(boolean isEnabledChangeStation, UserStationsDTO userStationsDTO) {
		if(!isEnabledChangeStation && userStationsDTO.getStations().isEmpty() && userStationsDTO.getDefaultStation() == null) {
			throw MesApplicationException.builder().exceptionInfo(StationServiceErrors.StationErrors.PRIVILEGE_CONFLICT.getExceptionInfo()).build();
		}
	}

	public interface StationServiceErrors{

		@Getter
		public enum StationErrors{
			PRIVILEGE_CONFLICT("PRIVILEGE_CONFLICT", "User doesn't have a valid default station and isn't enabled to change station");

			private StationErrors(String code, String description) {
				exceptionInfo = new ExceptionInfo(code, description);
			}

			private ExceptionInfo exceptionInfo;
		}
	}
}
