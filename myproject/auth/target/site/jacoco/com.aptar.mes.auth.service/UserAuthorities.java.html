package com.aptar.mes.auth.service;

import org.springframework.security.core.GrantedAuthority;
import org.springframework.security.core.authority.SimpleGrantedAuthority;

import lombok.Getter;

@Getter
public enum UserAuthorities {
	
	CHANGE_PASSWORD_AUTH("isEnabledChangePassword"),
	CHANGE_STATIONS_AUTH("isEnabledChangeStation");

	private UserAuthorities(String privilege) {
		grantedAuthority = new SimpleGrantedAuthority(privilege);
	}
	
	private GrantedAuthority grantedAuthority;
}
