package com.aptar.mes.auth.controller;

import java.security.Principal;
import java.util.logging.Level;

import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.http.HttpStatus;
import org.springframework.http.MediaType;
import org.springframework.security.core.Authentication;
import org.springframework.web.bind.annotation.GetMapping;
import org.springframework.web.bind.annotation.PathVariable;
import org.springframework.web.bind.annotation.RequestMapping;
import org.springframework.web.bind.annotation.ResponseStatus;
import org.springframework.web.bind.annotation.RestController;

import com.aptar.mes.auth.controller.response.UserStationsResponse;
import com.aptar.mes.auth.service.StationService;
import com.aptar.mes.auth.service.UserAuthorities;
import com.aptar.mes.auth.service.dto.StationDTO;

import io.swagger.annotations.Api;
import io.swagger.annotations.ApiOperation;
import io.swagger.annotations.ApiParam;
import io.swagger.annotations.ApiResponse;
import io.swagger.annotations.ApiResponses;
import lombok.extern.java.Log;

@RestController
@RequestMapping("/stations")
@Log
@Api(value = "station controller")
public class StationController {

	@Autowired
	private StationService service;

	@ApiOperation(value = "Get station list", response = StationDTO.class)
	@ApiResponses(value = { @ApiResponse(code = 200, message = "Success"),
			@ApiResponse(code = 400, message = "Bad Request"), @ApiResponse(code = 401, message = "Unauthorized"),
			@ApiResponse(code = 403, message = "Forbidden"), @ApiResponse(code = 404, message = "Not Found"),
			@ApiResponse(code = 500, message = "Internal Server Error") })
	@GetMapping(value = "/plant/{plantId}", produces = MediaType.APPLICATION_JSON_VALUE)
	@ResponseStatus(value = HttpStatus.OK)
	public UserStationsResponse findStationList(
			@ApiParam(value = "Plant id parameter for station list") @PathVariable String plantId, Principal principal,
			Authentication authentication) {
		log.log(Level.INFO, "Getting station list associated with the plant {0} for user {1} ", new Object[]{plantId, principal.getName()});

		return new UserStationsResponse(service.getStationList(plantId, principal.getName(), isEnabledChangeStation(authentication)));
	}

	private boolean isEnabledChangeStation(Authentication authentication) {
		return (authentication.getAuthorities() != null && authentication.getAuthorities()
				.contains(UserAuthorities.CHANGE_STATIONS_AUTH.getGrantedAuthority()));
	}

}
