package com.aptar.mes.auth.controller;

import java.security.Principal;
import java.util.logging.Level;

import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.http.HttpStatus;
import org.springframework.http.MediaType;
import org.springframework.security.core.Authentication;
import org.springframework.security.oauth2.provider.authentication.OAuth2AuthenticationDetails;
import org.springframework.security.oauth2.provider.token.TokenStore;
import org.springframework.web.bind.annotation.GetMapping;
import org.springframework.web.bind.annotation.RequestMapping;
import org.springframework.web.bind.annotation.ResponseStatus;
import org.springframework.web.bind.annotation.RestController;

import com.aptar.mes.auth.controller.response.ConfigurationsResponse;
import com.aptar.mes.auth.controller.response.UserPlantsResponse;
import com.aptar.mes.auth.service.PlantService;
import com.aptar.mes.auth.service.dto.PlantDTO;

import io.swagger.annotations.Api;
import io.swagger.annotations.ApiOperation;
import io.swagger.annotations.ApiResponse;
import io.swagger.annotations.ApiResponses;
import lombok.extern.java.Log;

@RestController
@RequestMapping("/plants")
@Log
@Api(value = "plant controller")
public class PlantController {
	
	@Autowired
	private PlantService service;
	
	@Autowired
	private TokenStore tokenStore;
	
	@ApiOperation(value = "Get plant list", response = PlantDTO.class)
	@ApiResponses(value = {
			@ApiResponse(code = 200, message = "Success"),
			@ApiResponse(code = 400, message = "Bad Request"),
			@ApiResponse(code = 401, message = "Unauthorized"),
			@ApiResponse(code = 403, message = "Forbidden"),
			@ApiResponse(code = 404, message = "Not Found"),
			@ApiResponse(code = 500, message = "Internal Server Error")
	})
	@GetMapping(produces=MediaType.APPLICATION_JSON_VALUE)
	@ResponseStatus(value=HttpStatus.OK)
	public UserPlantsResponse findPlantList(Principal principal) {
		
		log.log(Level.INFO, "Getting plant list associated with the user {0} ", principal.getName());
		return new UserPlantsResponse(service.getPlantList(principal.getName()));	
	}
	
	@ApiOperation(value = "Configurations")
	@ApiResponses(value = {
			@ApiResponse(code = 200, message = "Success"),
			@ApiResponse(code = 400, message = "Bad Request"),
			@ApiResponse(code = 401, message = "Unauthorized"),
			@ApiResponse(code = 403, message = "Forbidden"),
			@ApiResponse(code = 404, message = "Not Found"),
			@ApiResponse(code = 500, message = "Internal Server Error")
	})
	
	@GetMapping(value="/configurations", produces=MediaType.APPLICATION_JSON_VALUE)
	@ResponseStatus(value=HttpStatus.OK)
	public ConfigurationsResponse configurations(Authentication authentication)
	{
		OAuth2AuthenticationDetails oauthDetails = (OAuth2AuthenticationDetails) authentication.getDetails();
		String plantId = (String) tokenStore.readAccessToken(oauthDetails.getTokenValue()).getAdditionalInformation().get("plant");
		log.log(Level.INFO, "Getting configurations for plant {0}", new Object[] {plantId});

		return new ConfigurationsResponse(service.getPlantConfigurations(plantId));
	}
}
